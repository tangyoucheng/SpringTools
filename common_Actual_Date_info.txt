import os
import sys
import datetime
import numba
import numpy as np
import pandas as pd
import yfinance as yf
import exchange_calendars as xcals
import pytz  # 🚨 引入时区库，确保全球服务器运行都能锁定北京时间

# 使用免安装版本时，为了读取CDP_config.py，添加的设定
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

    
def get_stock_metrics_by_calendar(ticker: str, target_date: str) -> dict | None:
    """【资产管理级日历自适应核心组件 - 终极时区去伪存真版】

    1. 完美修正：全链路无死角锁死北京时间轴（Asia/Shanghai），海外服务器部署绝不发生未来函数时间穿透。
    2. 完美修正：强制列名大小写格式化，彻底封杀 yfinance 底层列名随机突变引发的 KeyError 崩溃。
    3. 完美修正：自适应 A股 15点未闭合日线强剪切，盘中运行自动安全退回上一闭合交易日，保证硬标尺幂等。
    """
    try:
        # 1. 🚨【核心修正：强行锁死北京时间轴】
        # 无论你的量化服务器托管在美国、新加坡还是本地，全量系统时钟严格向北京时间对齐！
        tz_bj = pytz.timezone("Asia/Shanghai")
        now_bj = datetime.datetime.now(tz_bj)

        # 统一解析输入的目标日期为纯净的无时区日期对象
        target_dt = pd.to_datetime(target_date).tz_localize(None)
        target_date_str = target_dt.strftime("%Y-%m-%d")

        # 2. 冗余窗口控制：往前狠取 12 天（对冲十一长假、春节长假断流），往后多取 1 天配合左闭右开
        start_date = target_dt - datetime.timedelta(days=12)
        end_date = target_dt + datetime.timedelta(days=1)

        # 3. 开启前复权下载历史行情
        df = yf.download(
            ticker,
            start=start_date.strftime("%Y-%m-%d"),
            end=end_date.strftime("%Y-%m-%d"),
            progress=False,
            auto_adjust=True,  # 强制开启前复权，消灭分红跳空虚假波幅
            timeout=5,  # 紧凑网络超时，防止多品种扫描时句柄卡死
        )

        if df.empty:
            print(
                f" 🚨 提示: yfinance 未能下载到 {ticker} 的历史数据或该标的已退市。"
            )
            return None

        # 4. 多重列索引解包降维保护
        if isinstance(df.columns, pd.MultiIndex):
            df.columns = df.columns.get_level_values(0)

        # 🚨【核心修正：列名大小写强行对齐】
        # 将列名强制格式化为首字母大写（High, Low, Close, Volume），彻底拍死 KeyError 隐患
        df.columns = [col.lower().capitalize() for col in df.columns]

        # 洗净 index 的时区魔鬼，转换为标准的十进制日期字符串
        df.index = (
            pd.to_datetime(df.index).tz_localize(None).strftime("%Y-%m-%d")
        )

        # 5. 🚨【核心修正：北京时间 15:00 盘中残缺K线切片剔除引擎】
        # 严格采信北京时间的今日日历
        bj_today_str = now_bj.strftime("%Y-%m-%d")
        last_k_date_str = str(df.index[-1])

        # 只有当行情源返回的最后一行确实触及了北京时间的今天，才激活时间轴判定
        if last_k_date_str == bj_today_str:
            # 严格以北京时间下午 15:00 收盘作为分水岭
            market_close_time_bj = now_bj.replace(
                hour=15, minute=0, second=0, microsecond=0
            )

            # 如果当前北京时间还没跨过 15:00，说明这根日K线是残缺的盘中半成品，必须狠心一刀切掉！
            if now_bj < market_close_time_bj:
                # 显式使用 .copy()，彻底拍死 SettingWithCopy 内存指针警告
                df = df.iloc[:-1].copy()

        # 6. 精准历史硬性时间戳拦截：剪掉大于目标日期的所有未来K线
        valid_df = df[df.index <= target_date_str]

        if valid_df.empty:
            print(
                f" 🚨 错误: 在下载的数据中，未找到小于或等于 {target_date_str} 的有效交易日。"
            )
            return None

        # 7. 锁定最终完全定型的交易日行指针
        actual_date_str = str(valid_df.index[-1])
        target_row = valid_df.iloc[-1]

        # 🚨【核心修正：高频多维解包防护层】
        # 针对潜在的 DataFrame 多重切片残留，执行最严厉的单元素标量化（Scalarization）提取
        close_v = target_row["Close"]
        high_v = target_row["High"]
        low_v = target_row["Low"]

        # 利用 numpy.squeeze 或 values 展平，誓死捍卫 float() 转换的确定性
        if hasattr(close_v, "values"):
            close_v = close_v.values
        if hasattr(high_v, "values"):
            high_v = high_v.values
        if hasattr(low_v, "values"):
            low_v = low_v.values

        # 如果提取出来依然是多维数组，取第一个元素
        if isinstance(close_v, (np.ndarray, pd.Series)):
            close_v = close_v.flatten()[0]
        if isinstance(high_v, (np.ndarray, pd.Series)):
            high_v = high_v.flatten()[0]
        if isinstance(low_v, (np.ndarray, pd.Series)):
            low_v = low_v.flatten()[0]

        # 8. 绝对无损包装，抛回给上层风控总线
        return {
            "Actual_Date": actual_date_str,  # 实际作为风控标尺计算基础的完全闭合交易日
            "Close": float(close_v),
            "High": float(high_v),
            "Low": float(low_v),
        }

    except Exception as e:
        print(f" 🚨 get_stock_metrics_by_calendar 运行异常: {e}")
        return None
