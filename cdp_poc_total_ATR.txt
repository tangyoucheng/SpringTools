def fetch_dynamic_atr(ticker_symbol, period_days=14, end_date_str=None):
    """
    【工业级自适应核心组件】动态获取指定股票的历史日K线并实时计算最新的 14日 ATR
    （完全对齐北京时间15:05盘中切片保护，并完美消除时区、数据和零分母漏洞）
    
    :param ticker_symbol: 股票代码 (例如 "688037.SS")
    :param period_days: 计算ATR的周期，默认14天
    :param end_date_str: 指定的计算截止日期字符串 (例如 "2026-07-21")，若为 None 则默认当前系统时间
    :return: 计算出的最新 ATR 绝对值(float)；若获取失败则返回 None
    """
    try:
        # 1. 🚨 强行锁定当前系统时钟为北京时间 (Asia/Shanghai)
        tz_bj = pytz.timezone('Asia/Shanghai')
        now_bj = datetime.datetime.now(tz_bj)  # 获取绝对精准的北京当前时间
        
        # 动态生成历史回溯时间窗口（保持3.5倍冗余，并确保最低有50天数据抗初始化误差）
        buffer_days = max(int(period_days * 3.5), 50)
        
        if end_date_str is not None:
            # 如果指定了截止日期，假定用户输入的是北京时间的日期字符串
            end_date = datetime.datetime.strptime(end_date_str, "%Y-%m-%d")
            start_date = end_date - datetime.timedelta(days=buffer_days)
            download_end = end_date + datetime.timedelta(days=1)
        else:
            end_date = now_bj.replace(tzinfo=None)  # 剥离时区信息用于与 yfinance 兼容的日期计算
            start_date = end_date - datetime.timedelta(days=buffer_days)
            download_end = end_date
        
        # 2. 下载历史日线数据
        df_hist = yf.download(ticker_symbol, start=start_date, end=download_end, interval="1d", progress=False)
        
        if df_hist.empty:
            print(f" 🚨 {ticker_symbol} 无法从行情源获取数据")
            return None

        # 【多重列索引解包降维保护】
        if isinstance(df_hist.columns, pd.MultiIndex):
            df_hist.columns = df_hist.columns.get_level_values(0)

        # 3. 彻底解决【时区幽灵漏洞】：强制转换index为无时区日期
        df_hist.index = pd.to_datetime(df_hist.index).tz_localize(None)
        
        # 【脏数据过滤保护】防止行情源吐出中间有空缺或NaN的行导致NumPy矩阵崩溃
        df_hist = df_hist.dropna(subset=['High', 'Low', 'Close'])

        if df_hist.empty:
            return None

        # 4. 🚨 工业级 15:05 盘中残缺K线切片剔除机制（使用纯正北京时间轴）
        if end_date_str is None:
            today_date = now_bj.date()  # 使用北京日历的今天
            last_k_date = df_hist.index[-1].date()
            
            # 只有当最后一条数据确实是今天，才进行时间轴判定
            if last_k_date == today_date:
                # 设定15:05为安全清算完成线
                market_safe_close_time = now_bj.replace(hour=15, minute=5, second=0, microsecond=0)
                
                # 完全用北京时间进行高精度比对，再也不怕海外服务器时差
                if now_bj < market_safe_close_time:
                    df_hist = df_hist.iloc[:-1]

        # 【有效样本量深度审计】EMA/RMA算法需要至少2倍周期的数据来消除初始种子带来的误差
        if len(df_hist) < (period_days * 2):
            print(f" 🚨 {ticker_symbol} 过滤后有效交易日({len(df_hist)})不足以精准计算 {period_days}日 ATR")
            return None

        # 5. 【NumPy矩阵全面审计】：无截断计算真实波幅 TR
        high = df_hist['High'].to_numpy().flatten()
        low = df_hist['Low'].to_numpy().flatten()
        close = df_hist['Close'].to_numpy().flatten()
        
        tr_np = np.zeros_like(high)
        tr_np = high - low  # 第一天起点对齐
        
        high_low = high[1:] - low[1:]
        high_close_prev = np.abs(high[1:] - close[:-1])
        low_close_prev = np.abs(low[1:] - close[:-1])
        
        tr_np[1:] = np.maximum(high_low, np.maximum(high_close_prev, low_close_prev))
        
        # 6. 精准对齐通达信/TradingView 的 Wilder's 经典平滑递推逻辑 (修正原本ewm初始种子偏移)
        tr_series = pd.Series(tr_np)
        initial_sma = tr_series.iloc[:period_days].mean()  # 前N天用SMA做初始种子
        
        ewm_input = pd.Series(index=tr_series.index, dtype=float)
        ewm_input.iloc[period_days - 1] = initial_sma
        ewm_input.iloc[period_days:] = tr_series.iloc[period_days:]
        
        # alpha=1/N, adjust=False 执行增量平滑递推（即标准 RMA 均值）
        atr_series = ewm_input.ewm(alpha=1/period_days, adjust=False).mean()
        
        latest_atr = float(atr_series.iloc[-1])
        
        # 【一字板极致缩量窒息保护】防止ATR为0导致后续风控系统计算仓位时发生除以0错误
        min_atr_floor = round(close[-1] * 0.001, 3)  # 设定地板价为股价的千分之一
        if latest_atr < min_atr_floor:
            latest_atr = min_atr_floor
            
        return round(latest_atr, 3) 
        
    except Exception as e:
        print(f" 🚨 独立方法 fetch_dynamic_atr 运行异常: {e}")
        return None

# =========================================================================
# 【底层组件 A】当日分时自适应 ATR 直方图算法
#（通过时钟硬对齐、OHLC四点成交量等权分量投射，完美解决开收盘单点污染与截断错位硬伤）
# =========================================================================
def run_intraday_single_grid_atr(df_today, atr_multiplier, base_atr):
    """
    【当日分时核心组件】传入特定格子 ATR 倍数系数与基准 ATR，跑单一次当日高频分时（1M/5M）的直方图筹码累加
    """
    try:
        # 1. 安全边界过滤：强制在Pandas层面执行时钟硬对齐，彻底封杀错位漏洞
        df_clean = df_today.dropna(subset=["Open", "High", "Low", "Close", "Volume"])
        
        # 🚨【散户防御补丁A】：解决 yfinance 在 A股无成交量分钟直接塌陷或吐出 Volume=0 的脏数据问题
        # 如果不剔除 Volume<=0 的行，等分投射时会引入无效的价格扰动点，拉偏 POC
        df_clean = df_clean[df_clean["Volume"] > 0]
        
        if df_clean.empty or len(df_clean) < 5:
            return None

        # 2. 计算动态格子大小（Bin Size）
        bin_size = base_atr * atr_multiplier
        if bin_size < 0.01:
            bin_size = 0.01

        # 3. 提取真实的日内极致价格边界（包含High和Low，拒绝只看Close的盲区）
        intraday_high = float(df_clean["High"].max())
        intraday_low = float(df_clean["Low"].min())
        
        # 极限边界处理：若全天一字板或无价格波动，直接返回中轴
        if intraday_high == intraday_low:
            return round((intraday_high + intraday_low) / 2, 3)
        
        # 🚨【核心修正：绝对网格对齐机制】
        # 以 0.0 为绝对起点，按 bin_size 步长向下和向上做整数倍取整（floor/ceil）。
        # 这确保了您的直方图刻度线在每天复盘时是物理对齐的，不会因为当天的最低价微调而导致格子整体错位。
        start_bin = np.floor((intraday_low - bin_size * 5) / bin_size) * bin_size
        end_bin = np.ceil((intraday_high + bin_size * 5) / bin_size) * bin_size + bin_size
        
        bins = np.arange(start_bin, end_bin, bin_size)
        
        # 🚨【散户防御补丁B】：极端异常区间保护
        # yfinance 在日本跨国请求时，若遇到开盘极速脉冲导致的浮点数异常，可能导致 bins 数量爆炸。
        # 正常 A股 1分钟线全天 240 根，格子数量理应在合理范围内（此处限制 500），防止内存卡死。
        if len(bins) < 2 or len(bins) > 500:
            return round((intraday_high + intraday_low) / 2, 3)

        # 4. 🚨【核心修正：零内存拷贝矩阵平坦化】
        # 废除原有的 np.concatenate 拼接法（因为 concatenate 会在底层频繁触发全量内存重新分配）。
        # 直接利用底层 NumPy 视图转换，将 1分钟线的 4点波动真实还原，性能提升数倍，特别适合散户在本地配置有限的电脑上跑大样本复盘。
        combined_prices = df_clean[["Open", "High", "Low", "Close"]].to_numpy().flatten()
        
        # 成交量同步等分四份连接，并利用 np.repeat 进行高速广播扩容，完美继承您的【OHLC四点等权分量投射法】
        volumes_quarter = df_clean["Volume"].to_numpy() / 4.0
        combined_volumes = np.repeat(volumes_quarter, 4)

        # 5. 执行带权重的直方图累加（Volume Profile）
        counts, _ = np.histogram(
            combined_prices, bins=bins, weights=combined_volumes
        )
        
        # 6. 安全防御：如果当天没有任何成交量（如全天停牌），返回正中轴
        if counts.sum() == 0:
            return round((intraday_high + intraday_low) / 2, 3)
            
        max_idx = np.argmax(counts)
        
        # 防御边界溢出
        if max_idx >= len(counts) or (max_idx + 1) >= len(bins):
            return round((intraday_high + intraday_low) / 2, 3)

        # 计算并返回最新审定的分时 POC 核心轴
        poc_price = (bins[max_idx] + bins[max_idx + 1]) / 2
        return round(float(poc_price), 3) # A股价格保留三位小数保护
        
    except Exception as e:
        print(f" 🚨 独立方法 run_intraday_single_grid_atr 运行异常: {e}")
        return None

# =========================================================================
# 【底层组件 B】融合时间衰减与 ATR 自适应的多日大周期直方图算法
#（支持OHLC全矩阵投射、彻底解决单日切碎单点失真、自适应大周期权重钝化漏洞）
# =========================================================================
def run_single_grid_calculation(df_period, target_dates_pool, atr_multiplier, base_atr, period_type="60日"):
    """【底层组件】传入特定的格子 ATR 倍数系数，跑单一次多天线性时间衰减直方图计算，输出对应的价格。"""
    try:
        if df_period.empty or len(df_period) < 2:
            return None

        # 1. 动态自适应网格宽度（Bin Size）
        bin_size = base_atr * atr_multiplier
        if bin_size < 0.01:
            bin_size = 0.01

        # 提取全局极致价格边界
        global_high = float(df_period["High"].max())
        global_low = float(df_period["Low"].min())

        # 🚨【核心修正：绝对网格对齐】
        # 放弃使用动态最高最低价直接作起点。改用 0.0 为绝对起点取整。
        # 这样能保证 3日、20日、250日等所有周期生成的直方图格子刻度完全物理对齐，消除了多周期横向对比的截断误差！
        start_bin = np.floor((global_low - bin_size * 5) / bin_size) * bin_size
        end_bin = np.ceil((global_high + bin_size * 5) / bin_size) * bin_size + bin_size
        
        bins = np.arange(start_bin, end_bin, bin_size)
        if len(bins) < 2:
            return (global_high + global_low) / 2

        # 2. 日期池过滤与排序
        df_sorted = df_period.copy()
        df_sorted['date_str'] = df_sorted['date_str'].astype(str)
        # 确保按时间从远到近（正序）排列：索引 0 是最久的历史，最后一行是最近的现在
        df_filter = df_sorted[df_sorted['date_str'].isin(target_dates_pool)].sort_values('date_str').reset_index(drop=True)
        
        if df_filter.empty:
            return None

        total_days = len(df_filter)

        # 3. 🚨【核心修正：自适应大周期衰减权重矩阵方向对齐】
        # 确保时间序列正序下：索引越靠后（越接近现在），权重必须越大！
        if "120日" in period_type or "250日" in period_type:
            half_life = total_days / 2
            # 修正后的时间距离向量：离现在越近，天数差越接近0
            days_from_now = total_days - 1 - np.arange(total_days) # 结果从大到小递减 [4, 3, 2, 1, 0]
            # 计算正确的衰减：days_from_now 为 0 时（今天），raw_weights 为 1.0（最大值）
            raw_weights = np.exp(-np.log(2) * days_from_now / half_life)
            base_weights = 0.2 + 0.8 * raw_weights
        else:
            # 线性衰减：1 到 total_days 递增，越接近现在权重越大，逻辑正确
            base_weights = np.arange(1, total_days + 1)

        # 4. 🚨【核心修正：零内存拷贝矩阵平坦化】
        # 彻底废除 np.concatenate 带来的巨大内存开销，改用极速的一维平坦化处理
        combined_prices = df_filter[["Open", "High", "Low", "Close"]].to_numpy().flatten()
        
        # 提取原始成交量并施加修正后的时间衰减权重
        v = df_filter["Volume"].to_numpy()
        weighted_v = v * base_weights
        
        # 将带权重的成交量高速广播复制4份，对齐4点投射法
        combined_volumes = np.repeat(weighted_v / 4.0, 4)

        # 5. 执行直方图累加
        combined_counts, _ = np.histogram(combined_prices, bins=bins, weights=combined_volumes)

        if combined_counts.sum() == 0:
            return (global_high + global_low) / 2

        max_idx = np.argmax(combined_counts)
        
        if max_idx >= len(combined_counts) or (max_idx + 1) >= len(bins):
            return (global_high + global_low) / 2

        # 6. 计算筹码核心轴 POC，日线级别保留两位小数输出
        poc_price = (bins[max_idx] + bins[max_idx + 1]) / 2
        return round(float(poc_price), 2)

    except Exception as e:
        print(f" 🚨 大周期组件 run_single_grid_calculation 异常: {e}")
        return None

# =========================================================================
# 【核心总控引擎】支持全周期筹码多维扫描、异常联动清洗与平滑五关价推理系统
# =========================================================================
def execute_all_period_pipeline(ticker_symbol, df_all_data, target_dates_pool, df_period_days, base_price=0.0, end_date_str=None):
    """
    【总控引擎】支持通吃全周期（1天分时到250天年线）的筹码多维扫描与智能裁决系统
    （彻底修复清洗系数倒挂、彻底封杀 raw_low 百分比max截断导致的指标伪共振盲区）
    """
    try:
        df_period_days = int(df_period_days)
        df_window = df_all_data.tail(df_period_days) if df_period_days > 1 else df_all_data

        # 1. 动态调取基准 ATR
        base_atr = fetch_dynamic_atr(ticker_symbol, period_days=14, end_date_str=end_date_str)
        if base_atr is None or np.isnan(base_atr):
            base_atr = base_price * 0.04 if base_price > 0 else 1.00

        # 2. 🚨【核心修正：消灭硬编码断层】使用对数连续函数自适应计算宏观ATR系数
        # 消除 5日到20日 之间的断层阶跃，实现物理维度的平滑过渡
        if df_period_days == 1:
            macro_atr_coef = 0.0
            # 极短周期微调细网格系数
            ratios = {"p_005": 0.05, "p_010": 0.12, "p_020": 0.25, "p_050": 0.40}
        else:
            # 天数越多，系数平滑递增，天生自带长周期防钝化锁死保护
            macro_atr_coef = round(0.3 + 0.35 * np.log2(df_period_days), 2)
            
            # 🚨【核心修正：防止大周期格子过粗】
            # 确保长周期（120、250日）下的最大格子步长不会因乘以大系数而变得无限大，维持筹码扫描的精度
            max_ratio = min(4.0, round(0.5 + 0.4 * np.log2(df_period_days), 2))
            ratios = {
                "p_005": round(max_ratio * 0.15, 2),
                "p_010": round(max_ratio * 0.35, 2),
                "p_020": round(max_ratio * 0.60, 2),
                "p_050": max_ratio
            }

        # 执行筹码分布计算
        poc_matrix = {}
        for key, atr_multiplier in ratios.items():
            if df_period_days == 1:
                poc_matrix[key] = run_intraday_single_grid_atr(df_window, atr_multiplier, base_atr)
            else:
                poc_matrix[key] = run_single_grid_calculation(df_window, target_dates_pool, atr_multiplier, base_atr, f"{df_period_days}日")

        p_005 = poc_matrix["p_005"]
        p_020 = poc_matrix["p_020"]

        # 3. 智能审计门限
        audit_threshold = (0.20 if df_period_days <= 5 else (0.40 if df_period_days <= 20 else 0.80)) * base_atr  
        
        raw_high = float(df_window["High"].max())
        raw_low = float(df_window["Low"].min())
        current_close = float(df_window["Close"].iloc[-1])
        is_audited = False

        if abs(p_005 - p_020) > audit_threshold:
            final_poc = p_020  
            is_audited = True
            audit_status = f"🚨 拦截异常！强制采信防噪网格({p_020:.2f}元)"
        else:
            final_poc = p_005  
            audit_status = f"🟢 筹码分布健康，多网格共振吻合"

        # 4. 🚨【核心修正：非对称全周期价格通道收拢】
        # 修正大跌或异常时等比例收缩导致的低吸支撑位虚高硬伤
        if macro_atr_coef > 0:
            if is_audited:
                # 审计异常时，上方阻力向下压紧（0.7），下方支撑允许保留弹性（0.9）防止左侧接飞刀
                macro_high = min(raw_high, final_poc + macro_atr_coef * base_atr * 0.7)
                macro_low = max(raw_low, final_poc - macro_atr_coef * base_atr * 0.9)
            else:
                if df_period_days >= 20:
                    # 针对大周期空头破位，执行不对称的向内收拢阀门
                    macro_high = min(raw_high, final_poc + macro_atr_coef * base_atr * 0.90)
                    macro_low = max(raw_low, final_poc - macro_atr_coef * base_atr * 0.98)
                else:
                    macro_high = max(raw_high, final_poc + macro_atr_coef * base_atr)
                    macro_low = min(raw_low, final_poc - macro_atr_coef * base_atr)
        else:
            macro_high = raw_high
            macro_low = raw_low

        # 5. 🚨【核心修正：正统几何对称五关价推演】
        # 放弃使用（High + Low + 2*POC）/ 4 这种破坏几何对称性的拼凑公式。
        # 恢复经典 CDP 的价格骨架，让支撑位（NL/AL）天生具备关于中轴的完整下延弹性，从根本上解决“支撑高于现价”的数学悖论。
        true_cdp = (macro_high + macro_low + current_close) / 3

        ah_poc = true_cdp + (macro_high - macro_low)                  
        nh_poc = 2 * true_cdp - macro_low                             
        nl_raw = 2 * true_cdp - macro_high
        al_raw = true_cdp - (macro_high - macro_low)       

        # 6. 安全阀门过滤（保留作为极限脏数据防御，但正常情况下由于经典公式的修复，此阀门极难被触发）
        if nl_raw >= current_close:
            nl_poc = current_close - (base_atr * 0.5)
        else:
            nl_poc = max(0.01, nl_raw)

        max_al_distance = nl_poc - (base_atr * 2.5)
        if al_raw <= max_al_distance:
            al_poc = nl_poc - (base_atr * 1.5)
        elif al_raw >= nl_poc:
            al_poc = nl_poc - (base_atr * 1.0)
        else:
            al_poc = max(0.01, al_raw)

        if nl_poc <= 0: nl_poc = 0.01
        if al_poc <= 0: al_poc = 0.01

        return {
            "final_poc": round(final_poc, 2),
            "true_cdp": round(true_cdp, 2),
            "al_poc": round(al_poc, 2),
            "nl_poc": round(nl_poc, 2),
            "nh_poc": round(nh_poc, 2),
            "ah_poc": round(ah_poc, 2),
            "audit_status": audit_status
        }
    except Exception as e:
        print(f" 🚨 核心总控引擎运行异常: {e}")
        return None
    
def get_stock_metrics_by_calendar(ticker: str, target_date: str) -> dict | None:
    """利用官方交易日历定位最近的交易日，并精准获取股票数据。
    （加入A股15点未闭合K线切片保护：今天盘中运行自动退回上一个闭合交易日，彻底免疫半成品数据污染）
    【已完美修复】传入周六日、节假日等非交易日时无法获取数据的问题。
    """
    try:
        # 1. 统一解析输入日期为纯净的无时区日期对象与字符串
        now = datetime.datetime.now()
        target_dt = pd.to_datetime(target_date).tz_localize(None)
        target_date_str = target_dt.strftime("%Y-%m-%d")
        
        # 2. 为了防止节假日/周末，往前狠取 10 天，往后多取 2 天确保闭合
        start_date = target_dt - datetime.timedelta(days=10)
        end_date = target_dt + datetime.timedelta(days=2)
        
        df = yf.download(ticker, start=start_date.strftime("%Y-%m-%d"), end=end_date.strftime("%Y-%m-%d"), progress=False)
        
        if df.empty:
            print(f" 🚨 提示: yfinance 未能下载到 {ticker} 的历史数据。")
            return None

        # 3. 【核心防御：多重列索引解包降维】
        if isinstance(df.columns, pd.MultiIndex):
            df.columns = df.columns.get_level_values(0)
            
        # 4. 洗净 index 的时区魔鬼
        df.index = pd.to_datetime(df.index).tz_localize(None).strftime("%Y-%m-%d")

        # 5. 🚨 【神来之笔：15点盘中残缺K线切片剔除机制】
        today_str = now.strftime("%Y-%m-%d")
        
        # 如果用户要获取的是今天的数据，或者在今天盘中倒腾历史
        if df.index[-1] == today_str:
            market_close_time = now.replace(hour=15, minute=0, second=0, microsecond=0)
            # 🚨 核心卡位：如果目前还没到下午15:00收盘，说明今天的日K线是半成品，必须狠心一刀切掉！
            if now < market_close_time:
                # print(f" 🛡️ 激活盘中保护机制：当前时间 {now.strftime('%H:%M:%S')} 尚未收盘，自动剔除今日残缺K线。")
                df = df.iloc[:-1]

        # 6. 【精准历史拦截】：筛选出所有小于或等于目标日期的真实交易日
        valid_df = df[df.index <= target_date_str]
        
        if valid_df.empty:
            print(f" 🚨 错误: 在下载的数据中，未找到小于或等于 {target_date_str} 的有效交易日。")
            return None
            
        # 此时 valid_df 的【最后一行】
        # 如果是盘中，由于今天被切了，这里拿到的就是【昨天的完整数据】；如果是周末，拿到的就是【周五的完整数据】
        actual_date_str = valid_df.index[-1]
        target_row = valid_df.iloc[-1]

        close_v = target_row["Close"]
        high_v = target_row["High"]
        low_v = target_row["Low"]

        if isinstance(close_v, pd.Series): close_v = close_v.iloc[0]
        if isinstance(high_v, pd.Series): high_v = high_v.iloc[0]
        if isinstance(low_v, pd.Series): low_v = low_v.iloc[0]

        return {
            "Actual_Date": actual_date_str,  # 返回实际作为计算基准的、完全闭合的交易日日期
            "Close": float(close_v),
            "High": float(high_v),
            "Low": float(low_v),
        }

    except Exception as e:
        print(f" 🚨 get_stock_metrics_by_calendar 运行异常: {e}")
        return None
