def export_cdp_to_excel(
    periods_pool,
    results_board,
    stock_info,
    #target_stock,  # 股票代码
    #stock_name,  # 股票名称
    #stock_metrics_close_p,  # 当前最新价/收盘价
    #stock_metrics_high_p,  # 最高价
    #stock_metrics_low_p,  # 最低价
    end_date,
    template_path="export_cdp_poc_stock_temple.xlsm",
):
    """将股票基本数据与CDP/POC数据无缝写入Excel模板，并在A1输出end_date。
    
    【已修改】所有表格数据统一从第二列（B列）开始对齐写入。
    """

    # 1. 定义统一的 CDP 数据表头
    headers = [
        "周期类型",
        "智能审定真POC",
        "中轴(CDP)",
        "止损边界(AL)",
        "低吸支撑(NL)",
        "高抛阻力(NH)",
        "追多边界(AH)",
    ]

    # 2. 循环遍历数据池，将每一行存入二维列表中
    table_rows = []
    for days in periods_pool:
        res = results_board[days]
        if days <= 5:
            name_str = f"{days}天短线历史"
        else:
            name_str = (
                f"20日线(月度)"
                if days == 20
                else (
                    f"60日线(季度)"
                    if days == 60
                    else (f"120日线(半年)" if days == 120 else f"250日线(年度)")
                )
            )

        table_rows.append(
            [
                name_str,
                RoundingStyle.ROUND.apply(res["final_poc"]),  # 四舍五入
                RoundingStyle.ROUND.apply(res["true_cdp"]),   # 四舍五入
                RoundingStyle.FLOOR.apply(res["al_poc"]),     # 直接舍去
                RoundingStyle.CEIL.apply(res["nl_poc"]),      # 直接进位
                RoundingStyle.CEIL.apply(res["nh_poc"]),      # 直接进位
                RoundingStyle.FLOOR.apply(res["ah_poc"]),     # 直接舍去
            ]
        )

    # 将二维列表转为 DataFrame 备用
    excel_data = pd.DataFrame(table_rows, columns=headers)

    # ======================
    # 写入Excel模板
    # ======================

    # 加载模板，保留 VBA 宏
    wb = load_workbook(template_path, keep_vba=True)
    ws = wb["Sheet1"]

    # 清空旧数据
    ws.delete_rows(1, ws.max_row)

    # 💡 核心控制：定义全局起始列（2 代表 B列）
    START_COL = 2
    START_ROW = 2

    # 【第一行第一列】：输出结束日期 end_date（第一行的其余列保持空白）
    # 依然保持在 A1，不受 START_COL 影响
    ws.cell(row=START_ROW, column=START_COL, value=end_date)
    START_ROW = START_ROW + 1;

    # 【第二行】：输出追加的股票属性表头（从第二列开始）
    # stock_info.keys() 会自动提取出 ["股票代码", "股票名称", "收盘价", "最高价", "最低价"]
    for col_idx, header_val in enumerate(stock_info.keys(), start=START_COL):
        ws.cell(row=START_ROW, column=col_idx, value=header_val)
    START_ROW = START_ROW + 1;

    # 【第三行】：输出对应的实际股票基本数据值（从第二列开始）
    # stock_info.values() 会自动提取出对应的实际数值（如 target_stock, stock_name 等）
    for col_idx, value_val in enumerate(stock_info.values(), start=START_COL):
        ws.cell(row=START_ROW, column=col_idx, value=value_val)
    START_ROW = START_ROW + 1;

    # 【第四行】：输出之前的 CDP/POC 数据的字段表头（从第二列开始）
    for col_idx, header in enumerate(excel_data.columns, start=START_COL):
        ws.cell(row=START_ROW, column=col_idx, value=header)
    START_ROW = START_ROW + 1;

    # 【第五行及以后】：依次无缝输出具体的 POC 数据行（从第二列开始）
    for row_offset, row in enumerate(
        excel_data.itertuples(index=False), start=START_ROW
    ):
        for col_idx, value in enumerate(row, start=START_COL):
            ws.cell(row=row_offset, column=col_idx, value=value)

    # 生成动态文件名
    excel_path = f"{stock_info["股票代码"]}_{stock_info["股票名称"]}_{end_date}.xlsm"

    # 保存文件
    wb.save(excel_path)
    print(f"{excel_path} 保存完成")
